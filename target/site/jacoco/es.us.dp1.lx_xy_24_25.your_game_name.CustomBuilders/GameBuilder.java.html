package es.us.dp1.lx_xy_24_25.your_game_name.CustomBuilders;

import org.jpatterns.gof.BuilderPattern;

import es.us.dp1.lx_xy_24_25.your_game_name.game.Game;
import es.us.dp1.lx_xy_24_25.your_game_name.user.User;

@BuilderPattern.ConcreteBuilder
public class GameBuilder implements Builder {
        
    private String gameName;
    private User creatorUser;
    private int playersNumber;
            
    public static Builder ofType(GameType gameType) {
        // En realidad no usamos TIPOS de partida en Game con un gameType, hemos creado el parámetro gameType
        // para mostrar que se podrían crear distintos tipos de partida desde el Builder
        return new GameBuilder();
    }
    
    @Override
    public Builder withGameName(String gameName) {
        this.gameName = gameName;
        return this;
    }

    @Override
    public Builder withCreatorUser(User creatorUser) {
        this.creatorUser = creatorUser;
        return this;
    }

    @Override
    public Builder withPlayersNumber(int playersNumber) {
        this.playersNumber = playersNumber;
        return this;
    }
    
    public Game build() {
            return new Game( gameName, creatorUser, playersNumber); 
    }
}
