package es.us.dp1.lx_xy_24_25.your_game_name.leader;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.http.HttpStatus;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.PathVariable;
import org.springframework.web.bind.annotation.PostMapping;
import org.springframework.web.bind.annotation.RequestBody;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RestController;

import io.swagger.v3.oas.annotations.security.SecurityRequirement;

@RestController
@RequestMapping("/api/v1/leaders")
@SecurityRequirement(name = "bearerAuth")
public class LeaderRestController {

    private final LeaderService leaderService;

    @Autowired
    public LeaderRestController(LeaderService leaderService) {
        this.leaderService = leaderService;
    }

    @GetMapping("/{idGame}/{roundNumber}/{missionNumber}")
    public Leader getLeader(@PathVariable Integer idGame, @PathVariable Integer roundNumber,
            @PathVariable Integer missionNumber) {
        return leaderService.getLeader(idGame, roundNumber, missionNumber);
    }

    @PostMapping
    public ResponseEntity<Leader> postLeader(@RequestBody LeaderDTO leaderDTO) {
        Leader newLeader = leaderService.postLeader(leaderDTO);
        return new ResponseEntity<>(newLeader, HttpStatus.CREATED);
    }
}
