package es.us.dp1.lx_xy_24_25.your_game_name.leader;

import jakarta.validation.constraints.Max;
import jakarta.validation.constraints.Min;
import jakarta.validation.constraints.NotNull;
import lombok.Getter;
import lombok.Setter;

@Getter
@Setter
public class LeaderDTO {
    @NotNull
    private Integer idGame; // ID del juego

    @NotNull
    @Min(1)
    @Max(5)
    private Integer roundNumber; // Número de ronda

    @NotNull
    @Min(1)
    @Max(5)
    private Integer missionNumber; // Número de misión

    @NotNull
    private Integer userLiderId; // ID del usuario líder (en lugar de la entidad completa)

    // Constructor
    public LeaderDTO(Integer idGame, Integer roundNumber, Integer missionNumber, Integer userLiderId) {
        this.idGame = idGame;
        this.roundNumber = roundNumber;
        this.missionNumber = missionNumber;
        this.userLiderId = userLiderId;
    }

    public LeaderDTO() {
    }

}
