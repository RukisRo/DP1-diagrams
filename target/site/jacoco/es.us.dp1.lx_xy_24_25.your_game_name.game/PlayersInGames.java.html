package es.us.dp1.lx_xy_24_25.your_game_name.game;

import jakarta.persistence.Entity;
import jakarta.persistence.JoinColumn;
import jakarta.persistence.ManyToOne;
import jakarta.persistence.Table;
import es.us.dp1.lx_xy_24_25.your_game_name.model.BaseEntity;
import es.us.dp1.lx_xy_24_25.your_game_name.user.User;
import lombok.Getter;
import lombok.Setter;

@Getter
@Setter
@Entity
@Table(name = "players_in_games")
public class PlayersInGames extends BaseEntity {

	@ManyToOne(optional = false)
	@JoinColumn(name = "id_player")	
	private User idPlayer; 

	@ManyToOne(optional = false)
	@JoinColumn(name = "id_game")	
	private Game idGame; 

	@ManyToOne(optional = true)
	@JoinColumn(name = "id_rol")	
	private Roles idRol; 


	// En este constructor, para asociar un Jugador a una Partida, no será necesario indicar
	// el Rol del jugador, porque al momento de unirse a la partida el rol aún no es conocido. 
	// Más tarde, cuando se asignen los Roles, se podrá actualizar el rol del jugador en esta partida.
	public PlayersInGames(User idPlayer, Game idGame) {
        this.idPlayer = idPlayer;
        this.idGame = idGame;
        
    }
	

	// Constructor sin parámetros
	public PlayersInGames() {
	}

	// Constructor con parámetros
	public PlayersInGames(User idPlayer, Game idGame, Roles idRol) {
		this.idPlayer = idPlayer;
		this.idGame = idGame;
		this.idRol = idRol; 
	}
}
