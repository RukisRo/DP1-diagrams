package es.us.dp1.lx_xy_24_25.your_game_name.game;

import jakarta.validation.constraints.Max;
import jakarta.validation.constraints.Min;
import jakarta.validation.constraints.NotNull;

public class GameCreationRequest {

    @NotNull
    private String gameName;  // Nombre de la partida

    @NotNull
    @Min(5)  // Número mínimo de jugadores
    @Max(10) // Número máximo de jugadores
    private int playersNumber;  // Número de jugadores en la partida

    @NotNull
    private int creatorUserId;  // ID del usuario creador

    // Getters y Setters
    public String getGameName() {
        return gameName;
    }

    public void setGameName(String gameName) {
        this.gameName = gameName;
    }

    public int getPlayersNumber() {
        return playersNumber;
    }

    public void setPlayersNumber(int playersNumber) {
        this.playersNumber = playersNumber;
    }

    public int getCreatorUserId() {
        return creatorUserId;
    }

    public void setCreatorUserId(int creatorUserId) {
        this.creatorUserId = creatorUserId;
    }
}
