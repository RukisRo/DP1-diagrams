package es.us.dp1.lx_xy_24_25.your_game_name.round;

import jakarta.validation.constraints.Max;
import jakarta.validation.constraints.Min;
import jakarta.validation.constraints.NotNull;

import lombok.Getter;
import lombok.Setter;

@Getter
@Setter
// Clase creada para modificar la representación proporcionada por la API, con sólo los valores necesarios
public class RoundDTO {

    public RoundDTO( Integer idGame, Integer idUser, Integer roundNumber, Integer missionNumber, Boolean vote) {
        this.idGame = idGame;
        this.idUser = idUser;
        this.roundNumber = roundNumber;
        this.missionNumber = missionNumber;
        this.vote = vote;
    }

    public RoundDTO() {
    }
    
    @NotNull
    private Integer idGame;

    @NotNull
    private Integer idUser;

    @NotNull
    @Min(1)
    @Max(5)
    private Integer roundNumber;
            
    @NotNull
    @Min(1)
    @Max(5)
    private Integer missionNumber;

    @NotNull
    private Boolean vote; 
}
