package es.us.dp1.lx_xy_24_25.your_game_name.round;
import jakarta.validation.constraints.Max;
import jakarta.validation.constraints.Min;
import es.us.dp1.lx_xy_24_25.your_game_name.game.Game;
import es.us.dp1.lx_xy_24_25.your_game_name.model.BaseEntity;
import es.us.dp1.lx_xy_24_25.your_game_name.user.User;
import jakarta.persistence.*;
import jakarta.validation.constraints.NotNull;
import lombok.EqualsAndHashCode;
import lombok.Getter;
import lombok.Setter;

@Getter
@Setter
@Entity
@Table(name = "rounds")
@EqualsAndHashCode(callSuper=false)
public class Round extends BaseEntity {
    
    @NotNull
    @ManyToOne                          
    @JoinColumn(name = "idGame")
    private Game game; // Partida a la que está relacionada la ronda

    //Eliminado idUser porque no le vemos la utilidad

    @NotNull
    @ManyToOne                          
    @JoinColumn(name = "idUser")
    private User user; // Usuario que vota en la Ronda

    // Si queremos saber a nivel de Estadísticas quién fue el jugador Lider de cada ronda
    // No necesario para poder jugar la partida, el guardarlo
    // Lo dejamos para acordarlo en el grupo
    /*@NotNull
    @ManyToOne                          
    @JoinColumn(name = "idLeaderUser")
    private User leaderUser;*/

    // Número de ronda que se vota para conformar Equipo, podrá ir entre 1 y 5
    @NotNull
    @Min(1)
    @Max(5)
    private Integer roundNumber; 

    // Nº de misión a la que pertenece la Ronda en la partida, 
    // irá siempre entre 1 y 5, y no necesita ser un Id a ninguna Entidad como se diseñó en 
    // un principio con IdMission
    @NotNull
    @Min(1)
    @Max(5)
    private Integer missionNumber;    

    // Voto del usuario : true será Aprobar, false será Rechazar
    //@NotNull
    private Boolean vote; 

}
