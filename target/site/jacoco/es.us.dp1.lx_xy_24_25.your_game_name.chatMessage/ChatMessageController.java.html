package es.us.dp1.lx_xy_24_25.your_game_name.chatMessage;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.*;

import io.swagger.v3.oas.annotations.security.SecurityRequirement;

import java.util.List;

@RestController
@RequestMapping("/api/v1/chatMessages")
@SecurityRequirement(name = "bearerAuth")
public class ChatMessageController {
    
    private ChatMessageService chatMessagesService;

    @Autowired
    public ChatMessageController( ChatMessageService chatMessagesService) {
        this.chatMessagesService = chatMessagesService;
    }

    @GetMapping("/{idGame}")
    public ResponseEntity<List<ChatMessageDTO>> getMessagesByGameId(@PathVariable Integer idGame) {
                
        List<ChatMessageDTO> messageList = chatMessagesService.getMessagesByGameId(idGame);
                        
        return ResponseEntity.ok(messageList);
    }

    @PostMapping
    public ResponseEntity<Void> addMessage(@RequestBody ChatMessageDTO chatMessageDTO) {
        // Llamamos al servicio para guardar el mensaje en la base de datos
        chatMessagesService.saveMessage(chatMessageDTO);
    
        // Devolvemos una respuesta vac√≠a con estado 200 OK
        return ResponseEntity.ok().build();
    }

}
