package es.us.dp1.lx_xy_24_25.your_game_name.chatMessage;

import java.time.LocalDateTime;

import jakarta.validation.constraints.NotNull;

import lombok.Getter;
import lombok.Setter;

@Getter
@Setter
// Clase creada para modificar la representación proporcionada por la API, con sólo los valores necesarios
public class ChatMessageDTO {

    public ChatMessageDTO( Integer idGame, Integer idUser, String userName, String message, LocalDateTime messageDateTime) {
        this.idGame = idGame;
        this.idUser = idUser;
        this.userName = userName;
        this.message = message;
        this.messageDateTime = messageDateTime;
    }
    
    @NotNull
    private Integer idGame;

    @NotNull
    private Integer idUser;

    @NotNull
    private String userName;
            
    @NotNull
    private String message;

    @NotNull
    private LocalDateTime messageDateTime;
}
