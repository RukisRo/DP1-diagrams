package es.us.dp1.lx_xy_24_25.your_game_name.mission;
import jakarta.validation.constraints.Max;
import jakarta.validation.constraints.Min;
import es.us.dp1.lx_xy_24_25.your_game_name.game.Game;
import es.us.dp1.lx_xy_24_25.your_game_name.model.BaseEntity;
import es.us.dp1.lx_xy_24_25.your_game_name.user.User;
import jakarta.persistence.*;
import jakarta.validation.constraints.NotNull;
import lombok.EqualsAndHashCode;
import lombok.Getter;
import lombok.Setter;

@Getter
@Setter
@Entity
@Table(name = "missions")
@EqualsAndHashCode(callSuper=false)
public class Mission extends BaseEntity {
    
    @NotNull
    @ManyToOne                          
    @JoinColumn(name = "idGame")
    private Game game;

    // Nº de misión en la partida, irá siempre entre 1 y 5, 
    // y por tanto no necesita ser un Id a ninguna Entidad como se diseñó en un principio 
    // con IdMission
    @NotNull
    @Min(1)
    @Max(5)
    private Integer missionNumber; 

    @NotNull
    @ManyToOne                          
    @JoinColumn(name = "idUser")
    private User user; // Usuario del Equipo conformado que vota en la Misión

    // Voto del usuario : true será Éxito, false será Fracaso
    //@NotNull
    private Boolean vote; 
    
}
