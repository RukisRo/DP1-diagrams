package es.us.dp1.lx_xy_24_25.your_game_name.team;

import org.jpatterns.gof.StrategyPattern;
import org.springframework.stereotype.Component;

@Component
@StrategyPattern
public class StandardGameStrategy implements TeamSizeStrategy {
    @Override
    public int calculateTeamSize(int totalPlayers, int missionNumber) {
        // Validar el número de jugadores
        if (totalPlayers < 5 || totalPlayers > 10) {
            throw new IllegalArgumentException("Número de jugadores fuera de rango (5-10 jugadores): " + totalPlayers);
        }

        // Validar la misión
        if (missionNumber < 1 || missionNumber > 5) {
            throw new IllegalArgumentException("Número de misión fuera de rango (1-5 misiones)");
        }

        // Tabla de tamaños de equipo (filas: misiones, columnas: jugadores)
        int[][] teamSizeTable = {
                { 2, 2, 2, 3, 3, 3 }, // 1ª Misión
                { 3, 3, 3, 4, 4, 4 }, // 2ª Misión
                { 2, 4, 3, 4, 4, 4 }, // 3ª Misión
                { 3, 3, 4, 5, 5, 5 }, // 4ª Misión
                { 3, 4, 4, 5, 5, 5 } // 5ª Misión
        };

        // Cálculo de índices en la tabla
        // Índice para el número de jugadores, de 5 a 10 jugadores -> índice de 0 a 5
        int playerIndex = totalPlayers - 5; // (5 jugadores -> índice 0, 10 jugadores -> índice 5)

        // Índice para la misión, de 1 a 5 misiones -> índice de 0 a 4
        int missionIndex = missionNumber - 1; // (Misión 1 -> índice 0, Misión 5 -> índice 4)

        // Obtener el tamaño del equipo de la tabla
        return teamSizeTable[missionIndex][playerIndex];
    }
}
