package es.us.dp1.lx_xy_24_25.your_game_name.CustomValidators;

import org.jpatterns.gof.StrategyPattern;
import org.springframework.stereotype.Component;
import org.springframework.validation.Errors;
import org.springframework.validation.Validator;

import es.us.dp1.lx_xy_24_25.your_game_name.game.Game;


@Component
@StrategyPattern
public class GameValidatorContext implements Validator {
    
    private Validator validator;

    // Equivalente a setStrategy(Strategy strategy), pero lo llamamos setValidator()    
    public void setValidator(Validator validator) {
        // Equivalente a asignar la clase Estrategia adecuada : this.strategy = strategy
        this.validator = validator;
    }

    @SuppressWarnings("null")
    @Override
    public boolean supports(Class<?> clazz) {

        return Game.class.isAssignableFrom(clazz);        
    }

    @SuppressWarnings("null")
    @Override
    public void validate(Object obj, Errors errors) throws RuntimeException {

        validator.validate(obj, errors);
    }    
}
